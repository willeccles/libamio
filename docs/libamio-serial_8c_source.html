57 int Serial_Open(Serial_Device* device, Serial_Params* params) {
58  device->_fd = open(params->dev, O_RDWR | O_NOCTTY | O_SYNC | O_NONBLOCK); // no CTTY makes sure the tty doesn't start controlling this program
59  if (device->_fd < 0) {
60  return -1;
61  }
62 
63  device->_type = params->type;
64 
65  if (params->type == SERIAL_UART) {
66  if (0 != set_interface_attribs_raw(device->_fd, get_baud_value(params))) {
67  return -1;
68  }
69  } else if (params->type == SERIAL_RS485) {
70  if (0 != set_interface_attribs_rs485(device->_fd, params)) {
71  return -1;
72  }
73  } else {

90 static int set_interface_attribs_raw(int fd, SERIAL_BAUD speed) {
91  struct termios tty;
92 
93  if (tcgetattr(fd, &tty) < 0) {
94  return -1;
95  }
96 
97  cfmakeraw(&tty);
98 
99  cfsetospeed(&tty, BAUDS[speed]);
100  cfsetispeed(&tty, BAUDS[speed]);
101 
102  tty.c_cflag &= ~(PARENB | CSTOPB | CSIZE | CRTSCTS);
103  tty.c_cflag |= (CLOCAL | CREAD | CS8);
104 
105  tty.c_oflag &= ~OPOST;
106  tty.c_cc[VMIN] = 0;
107  tty.c_cc[VTIME] = 5;
108 
109  if (tcsetattr(fd, TCSANOW, &tty) != 0) {
110  return -1;
111  }
112  return 0;
113 }

115 static int set_interface_attribs_rs485(int fd, Serial_Params* params) {
116  struct serial_rs485 rs485conf;
117 
118  // enable RS485 mode
119  rs485conf.flags |= SER_RS485_ENABLED;
120 
121  // RTS inactive when sending
122  rs485conf.flags &= ~(SER_RS485_RTS_ON_SEND);
123  // change if user specified
124  rs485conf.flags |= params->rs485_RTSHighOnSend * (SER_RS485_RTS_ON_SEND);
125 
126  // RTS inactive after sending
127  rs485conf.flags &= ~(SER_RS485_RTS_AFTER_SEND);
128  // change if user specified
129  rs485conf.flags |= params->rs485_RTSHighAfterSend * (SER_RS485_RTS_AFTER_SEND);
130 
131  // don't receive while sending
132  rs485conf.flags &= ~(SER_RS485_RX_DURING_TX);
133 
134  if (0 != ioctl(fd, TIOCSRS485, &rs485conf)) {
135  return -1;
136  }
137 
138  return set_interface_attribs_raw(fd, get_baud_value(params));
139 }
